let asteroids = []; let stars = []; let gameRunning = true; let explosionFrame = 0; function createStars() { for (let i = 0; i < 100; i++) { stars.push({ x: Math.random() * canvas.width, y: Math.random() * canvas.height, size: Math.random() * 2 + 1, speed: Math.random() * 1 + 0.2 }); } } function drawStars() { ctx.fillStyle = "white"; for (let star of stars) { ctx.beginPath(); ctx.arc(star.x, star.y, star.size, 0, Math.PI * 2); ctx.fill(); // Make stars slowly move down star.y += star.speed; if (star.y > canvas.height) { star.y = 0; star.x = Math.random() * canvas.width; } } } function drawRocket() { ctx.fillStyle = "gray"; ctx.beginPath(); ctx.moveTo(rocket.x + rocket.width / 2, rocket.y); ctx.lineTo(rocket.x + rocket.width, rocket.y + rocket.height); ctx.lineTo(rocket.x, rocket.y + rocket.height); ctx.closePath(); ctx.fill(); // Rocket window ctx.fillStyle = "lightblue"; ctx.beginPath(); ctx.arc(rocket.x + rocket.width / 2, rocket.y + 20, 10, 0, Math.PI * 2); ctx.fill(); // Rocket fins ctx.fillStyle = "red"; ctx.fillRect(rocket.x - 10, rocket.y + rocket.height - 20, 15, 20); ctx.fillRect(rocket.x + rocket.width - 5, rocket.y + rocket.height - 20, 15, 20); drawFlame(rocket.x + rocket.width / 2, rocket.y + rocket.height, 20); } function drawFlame(x, y, size) { ctx.beginPath(); ctx.moveTo(x, y); ctx.lineTo(x + size / 2, y + size); ctx.lineTo(x - size / 2, y + size); ctx.closePath(); const gradient = ctx.createLinearGradient(x, y, x, y + size); gradient.addColorStop(0, "yellow"); gradient.addColorStop(0.5, "orange"); gradient.addColorStop(1, "red"); ctx.fillStyle = gradient; ctx.fill(); } function createAsteroid() { const size = Math.random() * 50 + 20; asteroids.push({ x: Math.random() * canvas.width, y: -size, size: size, speed: Math.random() * 3 + 2 }); } function drawAsteroid(asteroid) { ctx.fillStyle = "#A29B8D"; // Grayish space rock ctx.beginPath(); ctx.arc(asteroid.x, asteroid.y, asteroid.size / 2, 0, Math.PI * 2); ctx.fill(); } function updateAsteroids() { for (let i = 0; i < asteroids.length; i++) { asteroids[i].y += asteroids[i].speed; if (asteroids[i].y > canvas.height) { asteroids.splice(i, 1); } } } function drawExplosion(x, y, size) { ctx.fillStyle = "orange"; ctx.beginPath(); ctx.arc(x, y, size * explosionFrame / 10, 0, Math.PI * 2); ctx.fill(); ctx.fillStyle = "red"; ctx.beginPath(); ctx.arc(x, y, size * explosionFrame / 15, 0, Math.PI * 2); ctx.fill(); } function checkCollisions() { for (let asteroid of asteroids) { if ( rocket.x < asteroid.x + asteroid.size && rocket.x + rocket.width > asteroid.x && rocket.y < asteroid.y + asteroid.size && rocket.y + rocket.height > asteroid.y ) { gameRunning = false; explodeRocket(); } } } function explodeRocket() { let explosionInterval = setInterval(() => { ctx.clearRect(0, 0, canvas.width, canvas.height); drawStars(); drawExplosion(rocket.x + rocket.width / 2, rocket.y + rocket.height / 2, 50); explosionFrame++; if (explosionFrame > 10) { clearInterval(explosionInterval); showGameOverMessage(); } }, 100); } function showGameOverMessage() { ctx.fillStyle = "rgba(0, 0, 0, 0.7)"; // Dark transparent background ctx.fillRect(0, canvas.height / 3, canvas.width, 200); ctx.fillStyle = "white"; ctx.font = "50px Arial"; ctx.textAlign = "center"; ctx.fillText("GAME OVER", canvas.width / 2, canvas.height / 2 - 20); ctx.font = "25px Arial"; const restartMessage = navigator.platform.includes("Mac") ? "Press âŒ˜ + R to Restart" : "Press F5 to Restart"; ctx.fillText(restartMessage, canvas.width / 2, canvas.height / 2 + 40); } function gameLoop() { if (!gameRunning) return; ctx.clearRect(0, 0, canvas.width, canvas.height); drawStars(); drawRocket(); updateAsteroids(); checkCollisions(); for (let asteroid of asteroids) { drawAsteroid(asteroid); } requestAnimationFrame(gameLoop); } document.addEventListener("keydown", (event) => { if (event.key === "ArrowLeft" && rocket.x > 0) rocket.x -= rocket.speed; if (event.key === "ArrowRight" && rocket.x + rocket.width < canvas.width) rocket.x += rocket.speed; }); createStars(); setInterval(createAsteroid, 1000); gameLoop();
